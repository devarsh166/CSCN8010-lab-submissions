Line Graph¶

This graph illustrates the periodic behavior of the sine (sin(x)) and cosine (cos(x)) functions over the range of 0 to 4π radians.¶

In [2]:
x = np.arange(0, 4 * np.pi, 0.1)
y = np.sin(x)
z = np.cos(x)

# Create custom x-axis labels
x_labels = ['0', r'$\frac{\pi}{2}$', r'$\pi$', r'$\frac{3\pi}{2}$', r'$2\pi$', r'$\frac{5\pi}{2}$', r'$3\pi$', r'$\frac{7\pi}{2}$', r'$4\pi$']

fig, ax = plt.subplots()

ax.plot(x, y, label='sin(x)')
ax.plot(x, z, label='cos(x)')

#Add vertical lines at specific x-values up to 4
vertical_lines_x = [0, np.pi / 2, np.pi, 3 * np.pi / 2, 2 * np.pi, 5 * np.pi / 2, 3 * np.pi, 7 * np.pi / 2, 4 * np.pi]
for vline_x in vertical_lines_x:
    ax.axvline(vline_x, color='gray', linestyle='--', linewidth=0.5)


# Add horizontal line at y=0
ax.axhline(0, color='gray', linestyle='--', linewidth=0.5)

plt.xlabel('x values from 0 to 4')
plt.ylabel('sin(x) and cos(x)')
plt.title('Plot of sin and cos from 0 to 4')
plt.xticks(np.arange(0, 4 * np.pi + 0.1, np.pi / 2), x_labels)
plt.legend()
plt.show()

# data from United Nations World Population Prospects (Revision 2019)
# https://population.un.org/wpp/, license: CC BY 3.0 IGO
year = [1950, 1960, 1970, 1980, 1990, 2000, 2010, 2018]
population_by_continent = {
    'africa': [228, 284, 365, 477, 631, 814, 1044, 1275],
    'americas': [340, 425, 519, 619, 727, 840, 943, 1006],
    'asia': [1394, 1686, 2120, 2625, 3202, 3714, 4169, 4560],
    'europe': [220, 253, 276, 295, 310, 303, 294, 293],
    'oceania': [12, 15, 19, 22, 26, 31, 36, 39],
}

fig, ax = plt.subplots()
ax.stackplot(year, population_by_continent.values(),
             labels=population_by_continent.keys(), alpha=0.8)
ax.legend(loc='upper left', reverse=True)
ax.set_title('World population (1950-2020)')
ax.set_xlabel('Year')
ax.set_ylabel('Number of people (in millions)')

plt.show()

import plotly.graph_objects as go

import pandas as pd
from datetime import datetime

df = pd.read_csv('https://raw.githubusercontent.com/plotly/datasets/master/finance-charts-apple.csv')

fig = go.Figure(data=[go.Candlestick(x=df['Date'],
                open=df['AAPL.Open'],
                high=df['AAPL.High'],
                low=df['AAPL.Low'],
                close=df['AAPL.Close'])])

fig.update_layout(
    title='The Great Recession',
    yaxis_title='Apple Stock',
    shapes = [dict(
        x0='2016-12-09', x1='2016-12-09', y0=0, y1=1, xref='x', yref='paper',
        line_width=2)],
    annotations=[dict(
        x='2016-12-09', y=0.05, xref='x', yref='paper',
        showarrow=False, xanchor='left', text='Increase Period Begins')]
)

fig.show()
